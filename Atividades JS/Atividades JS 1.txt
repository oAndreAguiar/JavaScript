//Faça um algoritmo para ler um valor correspondente a um lado de um quadrado e mostrar a ÁREA.//


var Side, Area;


Side = Number(window.prompt('Defina o lado do quadrado:'));

//Construa um programa que receba uma temperatura em graus Fahrenheit, e que calcule e apresente a temperatura em graus Celsius. Considere que C° = 5 x (F°-32)/9


var TempF, Celsius;


TempF = Number(window.prompt('Um numero em Fº:'));
Celsius = 5 * ((TempF - 32) / 9);
window.alert('Em Celsius é:  ' + String(Celsius))





//Construa um programa o qual leia a altura de uma pessoa e calcule seu índice 
//IMC, cuja fórmula é: IMC =peso/altura²


var peso, altura, IMC;


peso = Number(window.prompt('Defina o peso em KG (exemplo 90.00)'));
altura = Number(window.prompt('Defina sua altura em M (exemplo 1.80)'));
window.alert('Seu IMC é:  ' + String(IMC == peso / (altura * altura)));
